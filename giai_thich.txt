def normalize_angle(angle):
  Chức năng: Chuyển góc từ dải 0-360 sang dải -180 đến +180.
  Lý do: LiDAR trả về góc 0-360 độ. Ví dụ góc 350 độ thực ra là -10 độ (hơi lệch trái so với chính diện). 
  Hàm này chuyển 350 thành -10 để dễ xử lý hơn.

  if angle > 180:
    Nếu góc lớn hơn 180 (ví dụ 270, 350...)
  return angle - 360
    Trừ đi 360 để đưa về số âm. Ví dụ: 270 - 360 = -90, 350 - 360 = -10.
  return angle
    Nếu góc đã nhỏ hơn hoặc bằng 180 thì giữ nguyên.


def filter_scan(scan_data):
  Chức năng: Lọc dữ liệu quét LiDAR, chỉ giữ lại các điểm nằm trong phạm vi -90 đến +90 độ (phía trước cảm biến, 
  trùng hướng nhìn camera).

  filtered = []
    Tạo danh sách rỗng để chứa kết quả sau khi lọc.
  for q, angle, dist in scan_data:
    Duyệt qua từng điểm. Mỗi điểm gồm: q (chất lượng tín hiệu), angle (góc), dist (khoảng cách tính bằng mm).
  norm = normalize_angle(angle)
    Chuẩn hóa góc về dải -180 đến +180.
  if -90 <= norm <= 90 and dist > 0:
    Chỉ giữ điểm nằm trong +-90 độ VÀ có khoảng cách hợp lệ (lớn hơn 0).
  filtered.sort(key=lambda x: x[1])
    Sắp xếp các điểm theo góc từ nhỏ đến lớn (từ trái sang phải).


def init_lidar():

  lidar = RPLidar(PORT, baudrate=115200, timeout=3)
    Tạo kết nối tới LiDAR. baudrate=115200 là tốc độ truyền dữ liệu (bit/giây). timeout=3 nghĩa là chờ tối đa 3 giây khi đọc dữ liệu.
  info = lidar.get_info()
    Lấy thông tin phần cứng LiDAR (model, serial number...).
  health = lidar.get_health()
    Kiểm tra tình trạng LiDAR (Good/Warning/Error).
  return lidar
    Trả về đối tượng lidar để dùng ở các hàm khác.


def init_realsense():
  Chức năng: Khởi tạo camera Intel RealSense D435.

  pipeline = rs.pipeline()
    Tạo "đường ống" dữ liệu từ camera. Pipeline quản lý luồng dữ liệu từ camera đến chương trình.
  config = rs.config()
    Tạo cấu hình cho camera.
  config.enable_stream(rs.stream.depth, 640, 480, rs.format.z16, 30)
    Bật luồng ảnh độ sâu: kích thước 640x480 pixel, định dạng z16 (mỗi pixel là số 16-bit biểu thị khoảng cách mm), 30 khung hình/giây.
  config.enable_stream(rs.stream.color, 640, 480, rs.format.bgr8, 30)
    Bật luồng ảnh màu: kích thước 640x480 pixel, định dạng BGR 8-bit (mỗi kênh màu 0-255), 30 khung hình/giây.
  profile = pipeline.start(config)
    Khởi động camera với cấu hình đã thiết lập.
  align = rs.align(rs.stream.color)
    Tạo bộ căn chỉnh: làm cho ảnh depth khớp chính xác với ảnh color (vì 2 camera nằm ở vị trí khác nhau trên thiết bị).
  return pipeline, align
    Trả về pipeline và bộ căn chỉnh để dùng sau.

def lidar_thread(lidar):
  Chức năng: Hàm chạy trên thread riêng, liên tục đọc dữ liệu từ LiDAR và cập nhật biến latest_filtered.
  Lý do tách thread: Nếu không tách, khi chương trình đang chờ người dùng click chuột, LiDAR sẽ ngừng đọc 
  và biểu đồ polar sẽ đứng im.

  global latest_filtered, lidar_running
    Khai báo sẽ thay đổi 2 biến toàn cục này.
  for scan in lidar.iter_scans():
    Đọc liên tục từng vòng quét của LiDAR. Mỗi vòng quét trả về danh sách các điểm.
  if not lidar_running: break
    Nếu cờ lidar_running bị tắt (False), dừng vòng lặp.
  filtered = filter_scan(scan)
    Lọc scan chỉ giữ +-90 độ.
  with scan_lock:
    "Khóa" lại trước khi ghi dữ liệu, đảm bảo main thread không đọc đúng lúc đang ghi.
  latest_filtered = filtered
    Cập nhật dữ liệu mới nhất.

    Lấy danh sách khoảng cách.

  fig.canvas.flush_events()
    Xử lý các sự kiện đang chờ (click, resize...) và vẽ lại biểu đồ.

def select_bar_points(scan_data, polar_ax, polar_line, polar_fig, count):
  Chức năng: Hiển thị biểu đồ scatter (biểu đồ phân tán) của các điểm LiDAR. 
  Cho phép người dùng click chọn điểm đầu và điểm cuối của thanh trắng trên dữ liệu LiDAR. 
  Polar plot vẫn tiếp tục cập nhật trong lúc chờ.

  angles = [a for _, a, _ in scan_data]
  dists = [d for _, _, d in scan_data]
    Tách riêng danh sách góc và khoảng cách.
  selected = []
    Danh sách lưu chỉ số (index) của 2 điểm được chọn.

  fig2, ax2 = plt.subplots(figsize=(10, 5), num="LiDAR Points")
    Tạo cửa sổ biểu đồ mới, kích thước 10x5 inch, đặt tên "LiDAR Points".
  ax2.scatter(angles, dists, c='blue', s=10)
    Vẽ tất cả điểm LiDAR lên biểu đồ. c='blue' là màu xanh, s=10 là kích thước chấm.
  fig2.canvas.manager.window.wm_geometry("+700+50")
    Đặt cửa sổ ở vị trí (700px từ trái, 50px từ trên) để không chồng lên polar.

  def on_click(event):
    Hàm con (nested function) xử lý khi click chuột trên biểu đồ scatter.

    if event.inaxes != ax2 or len(selected) >= 2: return
      Bỏ qua nếu click bên ngoài biểu đồ hoặc đã chọn đủ 2 điểm.

    a_range = max(angles) - min(angles)
    d_range = max(dists) - min(dists)
      Tính khoảng cách lớn nhất của góc và distance, dùng để chuẩn hóa.

    Vòng for tìm điểm LiDAR gần nhất với vị trí click:
      dist_sq = ((a - event.xdata) / a_range)**2 + ((d - event.ydata) / d_range)**2
        Tính "khoảng cách chuẩn hóa" giữa mỗi điểm LiDAR và vị trí click. 
        Chia cho range để 2 trục (góc và khoảng cách) có cùng tỷ lệ, không bị lệch.

    ax2.scatter(..., marker='x', linewidths=3)
      Vẽ dấu X to tại điểm được chọn. Xanh lá cho P1, đỏ cho P2.
    ax2.annotate(...)
      Ghi chú thông tin (góc, khoảng cách) bên cạnh điểm.

  def on_key(event):
    if event.key == 'enter' and len(selected) == 2:
      plt.close(fig2)
      Khi nhấn Enter và đã chọn đủ 2 điểm, đóng cửa sổ scatter.

  fig2.canvas.mpl_connect('button_press_event', on_click)
  fig2.canvas.mpl_connect('key_press_event', on_key)
    Đăng ký 2 hàm callback: on_click cho sự kiện click, on_key cho sự kiện bấm phím.

  while plt.fignum_exists(fig2.number):
    plt.pause(0.05)
    Vòng lặp chờ cho đến khi cửa sổ scatter bị đóng. plt.pause(0.05) vừa chờ 50ms vừa xử lý sự kiện GUI.
    update_polar(...)
      Trong lúc chờ, vẫn cập nhật biểu đồ polar để nó không bị đóng băng.

  Sau khi cửa sổ đóng:
  a1 = angles[selected[0]]
  a2 = angles[selected[1]]
    Lấy góc của 2 điểm đã chọn.
  start = min(a1, a2)
  end = max(a1, a2)
    Xác định phạm vi góc nhỏ nhất đến lớn nhất.
  bar = [(q, a, d) for q, a, d in scan_data if start <= a <= end]
    Lọc chỉ giữ các điểm LiDAR nằm trong phạm vi góc đã chọn. Đây chính là các điểm thuộc thanh trắng.


def capture_frames(pipeline, align):
  Chức năng: Chụp 1 cặp ảnh (màu + độ sâu) từ camera RealSense.

  frames = pipeline.wait_for_frames(timeout_ms=5000)
    Chờ tối đa 5 giây để nhận dữ liệu từ camera.
  aligned = align.process(frames)
    Căn chỉnh ảnh depth với ảnh color (cho 2 ảnh khớp pixel).
  depth_frame = aligned.get_depth_frame()
  color_frame = aligned.get_color_frame()
    Lấy riêng ảnh depth và ảnh color.
  depth_img = np.asanyarray(depth_frame.get_data())
  color_img = np.asanyarray(color_frame.get_data())
    Chuyển dữ liệu camera thành mảng numpy (dạng số) để xử lý.

def map_lidar_to_image(bar_points, p1, p2):
  Chức năng: Ánh xạ (mapping) các điểm LiDAR thuộc thanh trắng sang tọa độ pixel trên ảnh.
  Nguyên lý: Nội suy tuyến tính. Điểm LiDAR có góc nhỏ nhất ứng với P1 (đầu trái), 
              góc lớn nhất ứng với P2 (đầu phải). Các điểm ở giữa được chia đều.

  min_a = min(angles)
  max_a = max(angles)
    Tìm góc nhỏ nhất và lớn nhất trong dãy.

  t = (angle - min_a) / (max_a - min_a)
    Tính tỷ lệ t nằm trong khoảng 0 đến 1. Ví dụ: nếu angle nằm đúng giữa min và max thì t = 0.5.
  px = int(p1[0] + t * (p2[0] - p1[0]))
  py = int(p1[1] + t * (p2[1] - p1[1]))
    Nội suy vị trí pixel: khi t=0 thì px=P1, khi t=1 thì px=P2, khi t=0.5 thì px nằm giữa P1 và P2.

  mapped.append({'angle': angle, 'distance': dist, 'pixel_x': px, 'pixel_y': py})

def load_calibration_jsons(calibration_dir):
  Chức năng: Đọc tất cả file JSON calibration trong thư mục captured_data/calibration/ và gom lại thành 2 danh sách tọa độ.

  json_files = sorted(glob.glob(os.path.join(calibration_dir, "calibration_*.json")))
    glob.glob tìm tất cả file có tên khớp mẫu "calibration_*.json". sorted() sắp xếp theo tên.

  Với mỗi file JSON, đọc phần mapped_points và chuyển đổi:
    angle_rad = np.radians(angle_deg)
      Chuyển góc từ độ sang radian.
    x = distance_mm * np.sin(angle_rad)
      Tính tọa độ X (trục ngang) từ tọa độ cực. sin(angle) cho thành phần ngang.
    z = distance_mm * np.cos(angle_rad)
      Tính tọa độ Z (trục sâu, phía trước) từ tọa độ cực. cos(angle) cho thành phần sâu.

  Tại sao chuyển từ polar sang Cartesian?
    LiDAR trả về dạng cực (góc + khoảng cách). Để tính toán hình học với camera, cần chuyển sang dạng Descartes (x, z) giống hệ tọa độ camera.

  Kết quả: 2 mảng numpy:
    lidar_pts: mỗi dòng là [x, z] (tọa độ LiDAR đã chuyển sang mm)
    pixel_pts: mỗi dòng là [pixel_x, pixel_y] (tọa độ pixel trên ảnh)


def get_realsense_intrinsics():
  Chức năng: Lấy thông số nội tại (intrinsics) của camera RealSense.

  Intrinsics là gì?
    Là các thông số bên trong camera mô tả cách camera chiếu điểm 3D thành pixel 2D. Gồm:
    - fx, fy: tiêu cự (focal length) theo pixel. Số càng lớn, ảnh càng "zoom".
    - cx, cy: tâm ảnh (principal point). Thường gần giữa ảnh (320, 240 cho ảnh 640x480).
    - distortion coefficients: hệ số méo ống kính.

  K = np.array([[fx, 0, cx], [0, fy, cy], [0, 0, 1]])
    Ma trận intrinsics 3x3 (còn gọi là camera matrix). Công thức chiếu:
    u = fx * X/Z + cx
    v = fy * Y/Z + cy
    Trong đó (X,Y,Z) là tọa độ 3D, (u,v) là tọa độ pixel.

  Nếu không kết nối được camera, dùng giá trị mặc định của D435.


def method_homography(lidar_pts, pixel_pts):
  Chức năng: Tìm ma trận Homography H (3x3) ánh xạ tọa độ LiDAR 2D sang pixel 2D.

  Homography là gì?
    Là phép biến đổi phối cảnh 2D-2D. Với 1 điểm LiDAR [x, z], Homography tính ra pixel [u, v] bằng:
    [u', v', w'] = H * [x, z, 1]
    u = u'/w', v = v'/w'

  H, mask = cv2.findHomography(lidar_pts, pixel_pts, cv2.RANSAC, 5.0)
    cv2.findHomography: hàm OpenCV tìm ma trận H tốt nhất từ các cặp điểm.
    cv2.RANSAC: thuật toán loại bỏ điểm nhiễu (outlier). Nó thử nhiều tổ hợp điểm, tìm H phù hợp nhất.
    5.0: ngưỡng sai số tối đa (pixel) để coi 1 điểm là "đúng" (inlier).
    mask: mảng cho biết điểm nào là inlier (1) hay outlier (0).

  Tính reprojection error (sai số chiếu ngược):
    lidar_h = np.hstack([lidar_pts, np.ones((len(lidar_pts), 1))])
      Thêm cột 1 vào sau [x, z] thành [x, z, 1] (tọa độ homogeneous).
    projected = (H @ lidar_h.T).T
      Nhân ma trận H với tọa độ LiDAR để ra tọa độ pixel dự đoán. @ là phép nhân ma trận.
    projected = projected[:, :2] / projected[:, 2:3]
      Chia cho w' để ra tọa độ pixel thực (chuẩn hóa).
    errors = np.linalg.norm(projected - pixel_pts, axis=1)
      Tính khoảng cách Euclidean giữa pixel dự đoán và pixel thật. Đây là sai số.


def method_solvepnp(lidar_pts, pixel_pts, K, dist_coeffs):
  Chức năng: Tìm ma trận xoay R và vector dịch chuyển T giữa LiDAR và camera bằng phương pháp Perspective-n-Point.

  solvePnP là gì?
    Cho các cặp (điểm 3D, pixel 2D) và camera intrinsics K, tìm cách camera "nhìn" các điểm 3D đó (vị trí + hướng quay).

  lidar_3d[:, 0] = lidar_pts[:, 0] / 1000.0  (x, mm sang mét)
  lidar_3d[:, 1] = 0.0                        (y = 0, vì LiDAR 2D quét ngang)
  lidar_3d[:, 2] = lidar_pts[:, 1] / 1000.0  (z, mm sang mét)
    Chuyển điểm LiDAR 2D thành 3D: [x, 0, z]. y=0 vì LiDAR quét trên 1 mặt phẳng ngang.

  success, rvec, tvec = cv2.solvePnP(lidar_3d, pixel_2d, K, dist_coeffs, flags=method)
    rvec: rotation vector (3 số biểu diễn góc xoay, dạng Rodrigues)
    tvec: translation vector (3 số biểu diễn dịch chuyển [tx, ty, tz] tính bằng mét)

  R, _ = cv2.Rodrigues(rvec)
    Chuyển rotation vector (3 số) thành rotation matrix (ma trận 3x3). Dạng ma trận dễ dùng hơn.

  projected, _ = cv2.projectPoints(lidar_3d, rvec, tvec, K, dist_coeffs)
    Chiếu các điểm 3D lên ảnh bằng R, T, K đã tìm được, để so sánh với pixel thật.


def method_affine(lidar_pts, pixel_pts):
  Chức năng: Tìm ma trận Affine A (2x3) ánh xạ LiDAR 2D sang pixel 2D.

  Affine Transform là gì?
    Là phép biến đổi bảo toàn đường thẳng song song. Gồm: xoay, co giãn, nghiêng, dịch chuyển. Nhưng KHÔNG có phối cảnh (perspective).
    [u, v] = A * [x, z, 1]

  A, inliers = cv2.estimateAffine2D(lidar_pts, pixel_pts, method=cv2.RANSAC)
    Tìm ma trận A tốt nhất, dùng RANSAC loại outlier.

  So sánh 3 phương pháp:
    - Homography: linh hoạt nhất (có perspective), cần 4+ điểm
    - solvePnP: chính xác nhất (dùng camera intrinsics), nhưng cần điểm ở nhiều vị trí khác nhau
    - Affine: ổn định nhất với ít điểm, nhưng không có perspective

def lidar_to_pixel(angle_deg, distance_mm, method, matrix):
  Chức năng: Chuyển 1 điểm LiDAR (góc, khoảng cách) thành tọa độ pixel (u, v) trên ảnh.

  angle_rad = math.radians(angle_deg)
    Chuyển góc sang radian.
  x = distance_mm * math.sin(angle_rad)
  z = distance_mm * math.cos(angle_rad)
    Chuyển tọa độ cực sang Descartes (giống compute_calibration).

  Nếu dùng Homography:
    pt = np.array([x, z, 1.0])
      Tạo tọa độ homogeneous.
    pixel = matrix @ pt
      Nhân ma trận H.
    u = pixel[0] / pixel[2]
    v = pixel[1] / pixel[2]
      Chuẩn hóa bằng cách chia cho phần tử thứ 3 (w).

  Nếu dùng Affine:
    pixel = matrix @ pt
      Nhân ma trận A (2x3). Kết quả đã là [u, v] trực tiếp, không cần chia.


def dist_to_color(dist_mm, min_d=300, max_d=4000):
  Chức năng: Tính màu điểm dựa theo khoảng cách. Gần = xanh, xa = đỏ.

  ratio = np.clip((dist_mm - min_d) / (max_d - min_d), 0, 1)
    Tính tỷ lệ khoảng cách nằm trong dải [min_d, max_d], giới hạn trong 0-1.
    np.clip đảm bảo giá trị không vượt ra ngoài 0 và 1.
  Nếu ratio < 0.5 (gần): xanh lá dần chuyển sang vàng.
  Nếu ratio >= 0.5 (xa): vàng dần chuyển sang đỏ.
  return (0, g, r) ở định dạng BGR (OpenCV dùng BGR thay vì RGB).
